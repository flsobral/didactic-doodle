cmake_minimum_required(VERSION 3.21)
project(tc_runtime VERSION 0.1.0 LANGUAGES C CXX)

set(TC_PLATFORM "DESKTOP" CACHE STRING "Target platform: DESKTOP, ANDROID, IOS, WEB")
set(TC_BACKEND "SDL" CACHE STRING "Platform backend: SDL, ANDROID_NATIVE, IOS_NATIVE, GLFW, WINIT")
set(TC_GRAPHICS "CPU" CACHE STRING "Graphics API: CPU, OPENGL, METAL, VULKAN")
set(TC_RENDERER "SKIA" CACHE STRING "2D renderer: SKIA, NANOVG, BLEND2D, VELLO")
set(TC_SKIA_TARGET "skia" CACHE STRING "CMake target exported by the Skia package")
set(TC_SKIA_ROOT "" CACHE PATH "Root of a TotalCross Skia release (headers/ and libskia-<platform>.a)")
set(TC_SKIA_LIBRARY "" CACHE FILEPATH "Optional explicit path to a Skia static library")
set_property(CACHE TC_PLATFORM PROPERTY STRINGS DESKTOP ANDROID IOS WEB)
set_property(CACHE TC_BACKEND PROPERTY STRINGS SDL ANDROID_NATIVE IOS_NATIVE GLFW WINIT)
set_property(CACHE TC_GRAPHICS PROPERTY STRINGS CPU OPENGL METAL VULKAN)
set_property(CACHE TC_RENDERER PROPERTY STRINGS SKIA NANOVG BLEND2D VELLO)

function(tc_require value description)
  list(FIND ARGN "${value}" index)
  if(index EQUAL -1)
    message(FATAL_ERROR "Unsupported ${description} '${value}'. Supported values: ${ARGN}")
  endif()
endfunction()

tc_require("${TC_PLATFORM}" "TC_PLATFORM" DESKTOP ANDROID IOS WEB)
tc_require("${TC_BACKEND}" "TC_BACKEND" SDL ANDROID_NATIVE IOS_NATIVE GLFW WINIT)
tc_require("${TC_GRAPHICS}" "TC_GRAPHICS" CPU OPENGL METAL VULKAN)
tc_require("${TC_RENDERER}" "TC_RENDERER" SKIA NANOVG BLEND2D VELLO)

if(NOT TC_BACKEND STREQUAL "SDL")
  message(FATAL_ERROR "TC_BACKEND=${TC_BACKEND} is a planned stub, not an implementation yet. Select -DTC_BACKEND=SDL.")
endif()
if(NOT TC_RENDERER STREQUAL "SKIA")
  message(FATAL_ERROR "TC_RENDERER=${TC_RENDERER} is a planned stub, not an implementation yet. Select -DTC_RENDERER=SKIA.")
endif()
if(TC_GRAPHICS STREQUAL "METAL" OR TC_GRAPHICS STREQUAL "VULKAN")
  message(FATAL_ERROR "TC_GRAPHICS=${TC_GRAPHICS} is a planned stub, not an implementation yet. Select CPU or OPENGL.")
endif()
if(TC_PLATFORM STREQUAL "ANDROID" OR TC_PLATFORM STREQUAL "IOS")
  message(FATAL_ERROR "TC_PLATFORM=${TC_PLATFORM} requires its native backend, which is not implemented yet.")
endif()
if(TC_PLATFORM STREQUAL "WEB" AND NOT TC_GRAPHICS STREQUAL "OPENGL")
  message(FATAL_ERROR "TC_PLATFORM=WEB currently requires -DTC_GRAPHICS=OPENGL.")
endif()
if(TC_GRAPHICS STREQUAL "OPENGL")
  message(FATAL_ERROR "TC_GRAPHICS=OPENGL is reserved and its context adapter is not implemented yet. Select -DTC_GRAPHICS=CPU.")
endif()

find_package(SDL3 CONFIG QUIET)
if(NOT TARGET SDL3::SDL3)
  message(FATAL_ERROR "SDL3 is required by TC_BACKEND=SDL. Install SDL3 or set CMAKE_PREFIX_PATH to its CMake package.")
endif()
if(TC_SKIA_ROOT)
  if(NOT TC_SKIA_LIBRARY)
    if(APPLE)
      set(TC_SKIA_LIBRARY "${TC_SKIA_ROOT}/libskia-macos-arm64.a")
    elseif(UNIX)
      set(TC_SKIA_LIBRARY "${TC_SKIA_ROOT}/libskia-linux-x86_64.a")
    endif()
  endif()
  if(NOT EXISTS "${TC_SKIA_ROOT}/headers/modules/skia/include/core/SkCanvas.h" OR NOT EXISTS "${TC_SKIA_LIBRARY}")
    message(FATAL_ERROR "TC_SKIA_ROOT must contain headers/modules/skia/include/core/SkCanvas.h and a matching libskia archive. See README.md for the supported TotalCross release layout.")
  endif()
  add_library(tc_skia STATIC IMPORTED GLOBAL)
  set_target_properties(tc_skia PROPERTIES IMPORTED_LOCATION "${TC_SKIA_LIBRARY}" INTERFACE_INCLUDE_DIRECTORIES "${TC_SKIA_ROOT}/headers/modules/skia")
  find_package(PNG REQUIRED)
  find_package(ZLIB REQUIRED)
  target_link_libraries(tc_skia INTERFACE PNG::PNG ZLIB::ZLIB)
  if(APPLE)
    find_library(COCOA_FRAMEWORK Cocoa REQUIRED)
    find_library(COREFOUNDATION_FRAMEWORK CoreFoundation REQUIRED)
    find_library(COREGRAPHICS_FRAMEWORK CoreGraphics REQUIRED)
    find_library(CORETEXT_FRAMEWORK CoreText REQUIRED)
    find_library(IMAGEIO_FRAMEWORK ImageIO REQUIRED)
    find_library(METAL_FRAMEWORK Metal REQUIRED)
    find_library(QUARTZCORE_FRAMEWORK QuartzCore REQUIRED)
    target_link_libraries(tc_skia INTERFACE ${COCOA_FRAMEWORK} ${COREFOUNDATION_FRAMEWORK} ${COREGRAPHICS_FRAMEWORK} ${CORETEXT_FRAMEWORK} ${IMAGEIO_FRAMEWORK} ${METAL_FRAMEWORK} ${QUARTZCORE_FRAMEWORK})
  endif()
  set(TC_SKIA_LINK_TARGET tc_skia)
else()
  find_package(skia CONFIG QUIET)
  if(NOT TARGET ${TC_SKIA_TARGET})
    message(FATAL_ERROR "Skia is required by TC_RENDERER=SKIA. Supply -DTC_SKIA_ROOT=<TotalCross release directory>, or provide a Skia CMake package exporting target '${TC_SKIA_TARGET}'.")
  endif()
  set(TC_SKIA_LINK_TARGET ${TC_SKIA_TARGET})
endif()

add_library(tc_runtime
  src/runtime/tc_app.c
  src/runtime/tc_scheduler.c
  src/runtime/tc_event_queue.c
  src/graphics/tc_cpu_context.c
  src/backends/sdl/tc_sdl_backend.c
  src/renderers/skia/tc_skia_renderer.cpp)
target_include_directories(tc_runtime PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)
target_include_directories(tc_runtime PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
target_compile_features(tc_runtime PUBLIC c_std_11)
target_compile_features(tc_runtime PRIVATE cxx_std_17)
target_link_libraries(tc_runtime PRIVATE SDL3::SDL3 ${TC_SKIA_LINK_TARGET})
target_compile_definitions(tc_runtime PRIVATE
  TC_BUILD_GRAPHICS_CPU=$<STREQUAL:${TC_GRAPHICS},CPU>
  TC_BUILD_GRAPHICS_OPENGL=$<STREQUAL:${TC_GRAPHICS},OPENGL>)

add_executable(tc_demo examples/demo/demo_main.c examples/demo/demo_scene.c)
target_link_libraries(tc_demo PRIVATE tc_runtime)
if(UNIX AND NOT APPLE)
  target_link_libraries(tc_demo PRIVATE m)
endif()
if(UNIX AND NOT APPLE)
  target_link_libraries(tc_demo PRIVATE m)
endif()
set_target_properties(tc_demo PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/examples/demo")

message(STATUS "tc_runtime: platform=${TC_PLATFORM}; backend=${TC_BACKEND}; graphics=${TC_GRAPHICS}; renderer=${TC_RENDERER}")
